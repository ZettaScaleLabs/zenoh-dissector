cmake_minimum_required(VERSION 3.15)
project(packet-zenoh C)

# ---------------------------------------------------------------------------
# Wireshark plugin requirements
# ---------------------------------------------------------------------------

# Locate Wireshark headers and library.
# find_package(Wireshark) uses cmake config mode but on NixOS the dev and
# runtime outputs are split across two store paths, so the config file's
# imported target references a path that doesn't exist in the dev output.
# Work around by locating headers and lib directly.
# On Windows, Wireshark headers live in the source tree built by
# scripts/build-wireshark-windows.ps1; libs are in the build run/ dir.
if(WIN32)
    # Headers from the downloaded source tarball (build-wireshark-windows.ps1 step 2).
    # wireshark.h is generated by cmake build; use epan/proto.h to locate the tree.
    file(GLOB _WS_SRC_DIRS "C:/wsbuild/wireshark-*")
    list(FILTER _WS_SRC_DIRS EXCLUDE REGEX "libs")
    find_path(Wireshark_INCLUDE_DIRS epan/proto.h
        HINTS ${_WS_SRC_DIRS}
        REQUIRED)

    # ws_version.h and config.h are generated by cmake configure; create minimal stubs.
    # WIRESHARK_VERSION is passed via -DWIRESHARK_VERSION=X.Y.Z from the CI.
    if(NOT DEFINED WIRESHARK_VERSION)
        set(WIRESHARK_VERSION "4.6.4")
    endif()
    string(REPLACE "." ";" _WS_VER_LIST "${WIRESHARK_VERSION}")
    list(GET _WS_VER_LIST 0 _WS_MAJOR)
    list(GET _WS_VER_LIST 1 _WS_MINOR)
    list(GET _WS_VER_LIST 2 _WS_MICRO)
    file(WRITE "${CMAKE_BINARY_DIR}/ws_version.h"
        "#pragma once\n"
        "#define WIRESHARK_VERSION_MAJOR ${_WS_MAJOR}\n"
        "#define WIRESHARK_VERSION_MINOR ${_WS_MINOR}\n"
        "#define WIRESHARK_VERSION_MICRO ${_WS_MICRO}\n"
        "#define VERSION_MAJOR ${_WS_MAJOR}\n"
        "#define VERSION_MINOR ${_WS_MINOR}\n"
        "#define VERSION_MICRO ${_WS_MICRO}\n"
    )
    file(WRITE "${CMAKE_BINARY_DIR}/config.h"
        "#pragma once\n"
        "/* Minimal config.h stub for Windows plugin build */\n"
        "#define HAVE_CONFIG_H 1\n"
        "#ifndef WIN32_LEAN_AND_MEAN\n"
        "#define WIN32_LEAN_AND_MEAN\n"
        "#endif\n"
        "#include <winsock2.h>\n"
        "#include <ws2tcpip.h>\n"
        "#define HAVE_WINSOCK2_H 1\n"
        "#define HAVE_WS2TCPIP_H 1\n"
        "/* MSYS2 GLib headers use GCC __attribute__; suppress for MSVC */\n"
        "#ifdef _MSC_VER\n"
        "# ifndef __attribute__\n"
        "#   define __attribute__(x)\n"
        "# endif\n"
        "#endif\n"
    )
    # wireshark.h and ws_symbol_export.h are cmake-generated in Wireshark 4.6+ and
    # absent from the source tarball; stub them for the plugin-only build.
    file(WRITE "${CMAKE_BINARY_DIR}/wireshark.h"
        "#pragma once\n"
        "#include <stdbool.h>\n"
        "#include <stdint.h>\n"
        "#include <stdlib.h>\n"
        "#include \"ws_version.h\"\n"
        "#include \"ws_symbol_export.h\"\n"
        "#include \"ws_attributes.h\"\n"
    )
    file(WRITE "${CMAKE_BINARY_DIR}/ws_codepoints.h"
        "#pragma once\n"
        "/* Stub: ws_codepoints.h defines Unicode codepoint constants; not needed for plugin build */\n"
    )
    file(WRITE "${CMAKE_BINARY_DIR}/ws_posix_compat.h"
        "#pragma once\n"
        "/* Stub: ws_posix_compat.h provides POSIX-on-Windows shims; not needed for plugin build */\n"
    )
    file(WRITE "${CMAKE_BINARY_DIR}/ws_diag_control.h"
        "#pragma once\n"
        "/* Stub: diagnostic control macros; not needed for plugin build */\n"
        "#define DIAG_OFF(x)\n"
        "#define DIAG_ON(x)\n"
        "#define DIAG_OFF_PEDANTIC\n"
        "#define DIAG_ON_PEDANTIC\n"
        "#define DIAG_OFF_DEPRECATED\n"
        "#define DIAG_ON_DEPRECATED\n"
        "#define DIAG_OFF_FORMAT_NONLITERAL\n"
        "#define DIAG_ON_FORMAT_NONLITERAL\n"
        "#define DIAG_OFF_CAST_QUAL\n"
        "#define DIAG_ON_CAST_QUAL\n"
    )
    file(WRITE "${CMAKE_BINARY_DIR}/ws_log_defs.h"
        "#pragma once\n"
        "/* Stub: ws_log_defs.h log level/domain definitions; not needed for plugin build */\n"
        "typedef int ws_log_level_t;\n"
        "#define LOG_LEVEL_NONE    0\n"
        "#define LOG_LEVEL_NOISY   1\n"
        "#define LOG_LEVEL_DEBUG   2\n"
        "#define LOG_LEVEL_INFO    3\n"
        "#define LOG_LEVEL_MESSAGE 4\n"
        "#define LOG_LEVEL_WARNING 5\n"
        "#define LOG_LEVEL_CRITICAL 6\n"
        "#define LOG_LEVEL_ERROR   7\n"
        "#define WS_LOG_DOMAIN NULL\n"
    )
    file(WRITE "${CMAKE_BINARY_DIR}/ws_attributes.h"
        "#pragma once\n"
        "#ifndef WS_NORETURN\n"
        "# ifdef _MSC_VER\n"
        "#   define WS_NORETURN __declspec(noreturn)\n"
        "# else\n"
        "#   define WS_NORETURN __attribute__((noreturn))\n"
        "# endif\n"
        "#endif\n"
        "#ifndef WS_RETNONNULL\n"
        "# define WS_RETNONNULL\n"
        "#endif\n"
        "#ifndef WS_DEPRECATED_X\n"
        "# define WS_DEPRECATED_X(x)\n"
        "#endif\n"
        "#ifndef WS_PRINTF_FORMAT\n"
        "# define WS_PRINTF_FORMAT(x,y)\n"
        "#endif\n"
    )
    file(WRITE "${CMAKE_BINARY_DIR}/ws_symbol_export.h"
        "#pragma once\n"
        "#ifdef WS_BUILD_DLL\n"
        "# define WS_DLL_PUBLIC __declspec(dllexport)\n"
        "#else\n"
        "# define WS_DLL_PUBLIC __declspec(dllimport)\n"
        "#endif\n"
        "#define WS_DLL_PUBLIC_DEF __declspec(dllimport)\n"
        "#define WS_DLL_LOCAL\n"
    )

    # Import libraries from the choco-installed Wireshark application.
    # The Windows Wireshark installer ships libwireshark.dll; we generate
    # wireshark.lib from it via scripts/build-wireshark-windows.ps1.
    find_library(Wireshark_LIBRARIES NAMES wireshark libwireshark
        HINTS
            "C:/Program Files/Wireshark"
            "C:/Program Files (x86)/Wireshark"
        REQUIRED)
else()
    find_path(Wireshark_INCLUDE_DIRS wireshark.h
        PATH_SUFFIXES wireshark
        REQUIRED)
    find_library(Wireshark_LIBRARIES wireshark REQUIRED)
    find_library(Wireshark_WSUTIL_LIB wsutil)  # pulled transitively on Linux; needed explicitly on macOS
endif()
message(STATUS "Wireshark: include=${Wireshark_INCLUDE_DIRS} lib=${Wireshark_LIBRARIES}")

# GLib is a Wireshark dependency not always transitively exposed.
# On Windows: MSYS2 provides glib headers; PKG_CONFIG_EXECUTABLE points to MSYS2 pkgconf.
# On Linux/macOS: system pkg-config finds it from the system package.
find_package(PkgConfig REQUIRED)
pkg_check_modules(GLIB2 REQUIRED IMPORTED_TARGET glib-2.0)
set(_GLIB2_TARGET PkgConfig::GLIB2)

# ---------------------------------------------------------------------------
# Rust cdylib path (built separately with: cargo build -p zenoh-codec-ffi)
# Default to debug build; override with -DZENOH_CODEC_FFI_LIB=...
# ---------------------------------------------------------------------------

if(NOT DEFINED ZENOH_CODEC_FFI_LIB)
    if(WIN32)
        set(ZENOH_CODEC_FFI_LIB "${CMAKE_SOURCE_DIR}/target/debug/zenoh_codec_ffi.dll")
    elseif(APPLE)
        set(ZENOH_CODEC_FFI_LIB "${CMAKE_SOURCE_DIR}/target/debug/libzenoh_codec_ffi.dylib")
    else()
        set(ZENOH_CODEC_FFI_LIB "${CMAKE_SOURCE_DIR}/target/debug/libzenoh_codec_ffi.so")
    endif()
endif()

# ---------------------------------------------------------------------------
# C plugin target
# ---------------------------------------------------------------------------

add_library(packet-zenoh MODULE packet-zenoh.c)

# Wireshark plugins must not have a "lib" prefix
set_target_properties(packet-zenoh PROPERTIES PREFIX "")

# On macOS the Rust cdylib has install name @rpath/libzenoh_codec_ffi.dylib.
# Embed rpaths so the plugin finds it both in the cmake build dir and after
# being copied to the Wireshark plugin directory alongside the dylib.
if(APPLE)
    get_filename_component(_FFI_DIR "${ZENOH_CODEC_FFI_LIB}" DIRECTORY)
    set_target_properties(packet-zenoh PROPERTIES
        INSTALL_RPATH "@loader_path;${_FFI_DIR}"
        BUILD_WITH_INSTALL_RPATH TRUE
    )
endif()

target_include_directories(packet-zenoh PRIVATE
    ${Wireshark_INCLUDE_DIRS}
    ${CMAKE_SOURCE_DIR}
    $<$<BOOL:${WIN32}>:${CMAKE_BINARY_DIR}>
)

target_compile_options(packet-zenoh PRIVATE
    $<$<C_COMPILER_ID:GNU,Clang>:-Wall -Wextra -std=c11>
    $<$<C_COMPILER_ID:MSVC>:/W3>
)

target_link_libraries(packet-zenoh PRIVATE
    ${Wireshark_LIBRARIES}
    $<$<BOOL:${Wireshark_WSUTIL_LIB}>:${Wireshark_WSUTIL_LIB}>
    ${_GLIB2_TARGET}
    ${ZENOH_CODEC_FFI_LIB}
)

# ---------------------------------------------------------------------------
# Install to user plugin directory
# ---------------------------------------------------------------------------

if(NOT DEFINED WIRESHARK_PLUGIN_DIR)
    # Determine from Wireshark version
    execute_process(
        COMMAND tshark --version
        OUTPUT_VARIABLE TSHARK_VERSION_OUTPUT
        ERROR_QUIET
    )
    string(REGEX MATCH "([0-9]+\\.[0-9]+)" WS_MAJOR_MINOR "${TSHARK_VERSION_OUTPUT}")
    if(WIN32)
        set(WIRESHARK_PLUGIN_DIR "$ENV{APPDATA}/Wireshark/plugins/${WS_MAJOR_MINOR}/epan")
    else()
        set(WIRESHARK_PLUGIN_DIR "$ENV{HOME}/.local/lib/wireshark/plugins/${WS_MAJOR_MINOR}/epan")
    endif()
endif()

install(TARGETS packet-zenoh
        LIBRARY DESTINATION "${WIRESHARK_PLUGIN_DIR}"
        RUNTIME DESTINATION "${WIRESHARK_PLUGIN_DIR}")
